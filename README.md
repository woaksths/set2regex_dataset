# set2regex_dataset


## Install 
    $git clone https://github.com/0xnurl/fado-python3.git
    $python3 -m venv venv
    $source venv/bin/activate
    $pip install -r requirements.txt
    $python fado-python3/setup.py install

## Usage 
    $python generate_regex.py
    $python main.py
